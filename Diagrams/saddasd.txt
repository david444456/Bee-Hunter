#include <iostream>
#include <vector>

using namespace std;

    vector<int> reconstruir(vector<int> &v, vector<int> conteo){
    	int indice_conteo = 0;

    	for (int i = 0; i < v.size(); i++){
    	    while(conteo[indice_conteo] == 0){
    		    indice_conteo++;
        	}
        	v[i] = indice_conteo;
        	conteo[indice_conteo]--;
        }
    }

    vector<int> contar(vector<int> v){
	    vector<int> conteo(8, 0);
	    for (int i = 0; i < v.size(); i++){
	    	conteo[v[i]]++;
	    }
	    return conteo;
    }

    vector<int> countingSort(vector<int> &v){
	    vector<int> conteo = contar(v);
    	return reconstruir(v, conteo);
    }

int main()
{
    vector<int> vec = {3, 4, 2, 3, 3, 4, 5, 6, 1, 2, 0, 7};
    countingSort(vec);
    for(int i = 0; i < vec.size(); i++){
        cout << vec[i] << " ";
    }

    return 0;
}
